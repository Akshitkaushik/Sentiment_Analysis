Sub ChangeWrapTextToTopAndBottom()
    Dim shape As Shape
    Dim inlineShape As InlineShape
    
    ' Loop through InlineShapes (images within text)
    For Each inlineShape In ActiveDocument.InlineShapes
        inlineShape.ConvertToShape 'Convert inline images to floating shapes
        inlineShape.WrapFormat.Type = wdWrapTopBottom 'Set wrap to Top and Bottom
    Next inlineShape
    
    ' Loop through floating Shapes (images not inline)
    For Each shape In ActiveDocument.Shapes
        shape.WrapFormat.Type = wdWrapTopBottom 'Set wrap to Top and Bottom
    Next shape
End Sub
