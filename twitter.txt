flowchart TD
    A[Start] --> B{Check if RPA or Input File is Received}
    B -- No --> C[Stop bot and send email]
    B -- Yes --> D[Login to Easy Application]
    D --> E[Select Database]
    E --> F{Database Column Empty?}
    F -- Yes --> G[Stop bot and send email]
    F -- No --> H[Enter Code CHGR]
    H --> I{Plan Number Blank?}
    I -- Yes --> J[Stop bot and send email]
    I -- No --> K[Insert Plan Number]
    K --> L[Press F12]
    L --> M{Check if pop-up exists}
    M -- Exists --> N[Click OK]
    M -- No Pop-up --> O[Select Change Group Account Recovery Information, Press Enter]
    O --> P{Rule Appears Twice or More?}
    P -- Yes --> Q[Do not use rule with term date]
    P -- No --> R[Press F12 to bring up selection screen]
    R --> S[Press F12 to go into Maintain Combo Details]
    S --> T[Ensure description is displayed as expected]
    T --> U[Press F12 twice to check for exclusions]
    U --> V{Exclusions Present?}
    V -- No --> W[Add the exclusions]
    V -- Yes --> X[Press F12 to reach % of BASIS section]
    X --> Y[Use down arrow to find a clean row]
    Y --> Z[Insert 0, then press F6 to save]
    Z --> AA[Press F11 four times to return to the previous screen]
    AA --> AB[Use down arrow to reach Debit Method Order Rules, Press F12]
    AB --> AC[Press F12 to go to data insert columns]
    AC --> AD{Are the three standard entries there?}
    AD -- No --> AE[Add missing entries]
    AD -- Yes --> AF[Press Ctrl+F9 to go to assessment row]
    AF --> AG[Insert a new row using Ctrl+F12]
    AG --> AH[Insert Begin Date, End Date, Schedule Date]
    AH --> AI[Ensure schedule date ends with 00:00]
    AI --> AJ[Press Enter to move to Requested Amount]
    AJ --> AK[System should insert date into Effective column]
    AK --> AL[Insert amount from spreadsheet]
    AL --> AM[Press F6 to save]
    AM --> AN[Press F3 twice to return to CHGR]
    AN --> AO[Continue to next record]
    AO --> AP[End]
